SELECT   T.交易日期,  CASE    WHEN T.幣別 = 'TWD'    THEN T.付款金額    WHEN T.幣別 = 'USD'    THEN T.付款金額 * H.匯率    WHEN T.幣別 = 'JPY'    THEN T.付款金額 * H.匯率    WHEN T.幣別 = 'EUR'    THEN T.付款金額 * H.匯率    WHEN T.幣別 = 'HKD'    THEN T.付款金額 * H.匯率    ELSE NULL  END AS 金額,  T.幣別  FROM   交易明細 AS T LEFT  JOIN   匯率資料表 AS H  ON T.幣別 = H.幣別  WHERE    T.公司金鑰 = '6224'  AND  T.交易日期  BETWEEN  DATE('now', '-1 year')  AND  DATE('now');
