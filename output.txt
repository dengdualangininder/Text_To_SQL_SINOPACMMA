SELECT   SUM(    CASE      WHEN T1.幣別 = 'USD'      THEN T1.餘額      WHEN T1.幣別 = 'JPY'      THEN T1.餘額 * (         SELECT           匯率          FROM   匯率資料表          WHERE            幣別 = 'JPY'  AND  公司金鑰 = '6224'         ORDER BY           生效日期 DESC         LIMIT  1      )      WHEN T1.幣別 = 'EUR'      THEN T1.餘額 * (         SELECT           匯率          FROM   匯率資料表          WHERE            幣別 = 'EUR'  AND  公司金鑰 = '6224'         ORDER BY           生效日期 DESC         LIMIT  1      )      WHEN T1.幣別 = 'HKD'      THEN T1.餘額 * (         SELECT           匯率          FROM   匯率資料表          WHERE            幣別 = 'HKD'  AND  公司金鑰 = '6224'         ORDER BY           生效日期 DESC         LIMIT  1      )      WHEN T1.幣別 = 'TWD'      THEN T1.餘額 * (         SELECT           匯率          FROM   匯率資料表          WHERE            幣別 = 'TWD'  AND  公司金鑰 = '6224'         ORDER BY           生效日期 DESC         LIMIT  1      )      ELSE 0    END  ) / (     SELECT       匯率      FROM   匯率資料表      WHERE        幣別 = 'USD'  AND  公司金鑰 = '6224'     ORDER BY       生效日期 DESC     LIMIT  1  ) AS total_usd  FROM   帳戶餘額表 AS T1  WHERE    T1.公司金鑰 = '6224';
